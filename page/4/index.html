栈队列二叉树的简单实现

#include
#include
using namespace std;
typedef char Elemtype;
typedef struct BiTNode{
Elemtype data;
BiTNode lchild,rchild;
}BiTNode,BiTree;
int CreatBiTree(BiTree &T){ //建立二叉树
Elemtype ch;
cin>>ch;
if(ch==’#’) T=NULL;
else{
T=(BiTNode)malloc(sizeof(BiTNode));
T->data=ch;
CreatBiTree(T->lchild);
CreatBiTree(T->rchild);
}
return 0;
}
void PreOrder(BiTree &T){
if(T!=NULL){
cout<data<<endl;
PreOrder(T->lchild);
PreOrder(T->rchild);
}
}
void InOrder(BiTree &T){
if(T!=NULL){
InOrder(T->lchild);
cout<data<<endl;
InOrder(T->rchild);
}
}
void PostOrder(BiTree &T){
if(T!=NULL){
PostOrder(T->lchild);
PostOrder(T->rchild);
cout<data<<endl;
}
}

int main(){
BiTree T;
CreatBiTree(T);
printf(“先序遍历:\n”); // a
PreOrder(T); // b c
printf(“中序遍历:\n”); //d e
InOrder(T); // f g
printf(“后序遍历:\n”);
PostOrder(T);
return 0;
}

数据结构几种排序的简单实现

源代码如下,亲测可用(基于c语言)
#include<stdio.h>
#include<stdlib.h>
void swap(int a,int *b){ //交换
int temp;
temp=a;
a=b;
*b=temp;
}
//直接插入排序
void Insert(int a[],int n){
int i=0,j=0,k=0,temp=0;
for(i=1;i<n;i++){
for(j=0;j<i;j++){
if(a[i]<a[j]){
temp=a[i];
for(k=i-1;k>=j;k–){
a[k+1]=a[k];
}
a[j]=temp;
}
}
}
}
//希尔排序
void Shell(int a[],int n){
int D=5,Dk=0,i=0,j=0,k=0,temp=0;
for(D=5;D>=1;D/=2){ //增量
for(Dk=0;Dk<D;Dk++){ //执行增量排序次数
for(i=Dk+D;i<n;i+=D){ //基于增量的直接插入排序
for(j=Dk;j<i;j+=D){
if(a[i]<a[j]){
temp=a[i];
for(k=i-D;k>=j;k-=D){
a[k+D]=a[k];
}
a[j]=temp;
}
}
}
}
}
}
//简单选择排序
void Select(int a[],int n){
int i=0,j=0,min=-1,minIndex,temp=0;
for(i=0;i<n-1;i++){
min=a[i];
for(j=i;j<n;j++){ //找出最小元素的下标
if(a[j]<min){
min=a[j];
minIndex=j;
}
}
swap(a+i,a+minIndex);
}
}
//冒泡排序
void Bubble(int a[],int n){
int i=0,j=0,temp=0;
for(i=n-1;i>0;i–){
for(j=0;j<i;j++){
if(a[j]>a[j+1]){
swap(a+j,a+j+1);
}
}
}
}
//快速排序
void Quick(int a[],int low,int high){
if(low>high) return;
int L=low,R=high,pivot=a[low];
while(L<R){
while(L<R&&a[R]>=pivot){ //从右边开始找到比枢纽小的值移到左边
R–;
}
if(L<R){
a[L]=a[R];
L++;
}
while(L<R&&a[L]<pivot){ //从左边开始找到比枢纽小的值移到右边
L++;
}
if(L<R){
a[R]=a[L];
R–;
}
}
Quick(a,low,L-1);
Quick(a,L+1,high);

}
