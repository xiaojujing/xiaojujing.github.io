栈队列二叉树的简单实现

#include
#include
using namespace std;
typedef char Elemtype;
typedef struct BiTNode{
Elemtype data;
BiTNode lchild,rchild;
}BiTNode,BiTree;
int CreatBiTree(BiTree &T){ //建立二叉树
Elemtype ch;
cin>>ch;
if(ch==’#’) T=NULL;
else{
T=(BiTNode)malloc(sizeof(BiTNode));
T->data=ch;
CreatBiTree(T->lchild);
CreatBiTree(T->rchild);
}
return 0;
}
void PreOrder(BiTree &T){
if(T!=NULL){
cout<data<<endl;
PreOrder(T->lchild);
PreOrder(T->rchild);
}
}
void InOrder(BiTree &T){
if(T!=NULL){
InOrder(T->lchild);
cout<data<<endl;
InOrder(T->rchild);
}
}
void PostOrder(BiTree &T){
if(T!=NULL){
PostOrder(T->lchild);
PostOrder(T->rchild);
cout<data<<endl;
}
}

int main(){
BiTree T;
CreatBiTree(T);
printf(“先序遍历:\n”); // a
PreOrder(T); // b c
printf(“中序遍历:\n”); //d e
InOrder(T); // f g
printf(“后序遍历:\n”);
PostOrder(T);
return 0;
}
