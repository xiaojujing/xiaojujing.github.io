}

int total=0;
priority_queue<Stall> pq;
for(int i=0;i<n;++i){
    if(pq.empty()){

            pos[cows[i].No]=st.No;
            pq.push(Stall(cows[i].b,st.No));

            pq.push(Stall(cows[i].b,total));
