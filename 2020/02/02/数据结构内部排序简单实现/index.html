数据结构几种排序的简单实现

源代码如下,亲测可用(基于c语言)
#include<stdio.h>
#include<stdlib.h>
void swap(int a,int *b){ //交换
int temp;
temp=a;
a=b;
*b=temp;
}
//直接插入排序
void Insert(int a[],int n){
int i=0,j=0,k=0,temp=0;
for(i=1;i<n;i++){
for(j=0;j<i;j++){
if(a[i]<a[j]){
temp=a[i];
for(k=i-1;k>=j;k–){
a[k+1]=a[k];
}
a[j]=temp;
}
}
}
}
//希尔排序
void Shell(int a[],int n){
int D=5,Dk=0,i=0,j=0,k=0,temp=0;
for(D=5;D>=1;D/=2){ //增量
for(Dk=0;Dk<D;Dk++){ //执行增量排序次数
for(i=Dk+D;i<n;i+=D){ //基于增量的直接插入排序
for(j=Dk;j<i;j+=D){
if(a[i]<a[j]){
temp=a[i];
for(k=i-D;k>=j;k-=D){
a[k+D]=a[k];
}
a[j]=temp;
}
}
}
}
}
}
//简单选择排序
void Select(int a[],int n){
int i=0,j=0,min=-1,minIndex,temp=0;
for(i=0;i<n-1;i++){
min=a[i];
for(j=i;j<n;j++){ //找出最小元素的下标
if(a[j]<min){
min=a[j];
minIndex=j;
}
}
swap(a+i,a+minIndex);
}
}
//冒泡排序
void Bubble(int a[],int n){
int i=0,j=0,temp=0;
for(i=n-1;i>0;i–){
for(j=0;j<i;j++){
if(a[j]>a[j+1]){
swap(a+j,a+j+1);
}
}
}
}
//快速排序
void Quick(int a[],int low,int high){
if(low>high) return;
int L=low,R=high,pivot=a[low];
while(L<R){
while(L<R&&a[R]>=pivot){ //从右边开始找到比枢纽小的值移到左边
R–;
}
if(L<R){
a[L]=a[R];
L++;
}
while(L<R&&a[L]<pivot){ //从左边开始找到比枢纽小的值移到右边
L++;
}
if(L<R){
a[R]=a[L];
R–;
}
}
Quick(a,low,L-1);
Quick(a,L+1,high);

}

int begin=0,end=0;
int i=0,a[100]={0};
for(i=0;i<100;i++){
a[i]=rand()%1000;
}
//Insert(a,100);

Quick(a,0,99);
for(i=1;i<=100;i++){
printf(“%d “,a[i-1]);
if(i%10==0) printf(“\n”);
}
return 0;
}
